// 括号表达式测试用例
// 测试功能：
// 1. 在表达式中使用括号，验证 ParenthesizedExpression 的正确处理
// 2. 测试括号表达式在变量初始化中的使用
// 3. 测试括号表达式在条件判断中的使用
// 4. 测试嵌套括号表达式

int main() {
    int a = 10;
    int b = 20;
    int c = 10;
    
    // 测试括号表达式在变量初始化中
    int eq_result = (a == c);      // 10 == 10 -> 1
    int ne_result = (a != b);       // 10 != 20 -> 1
    int lt_result = (a < b);        // 10 < 20 -> 1
    int le_result = (a <= c);       // 10 <= 10 -> 1
    int gt_result = (b > a);        // 20 > 10 -> 1
    int ge_result = (c >= a);       // 10 >= 10 -> 1
    
    // 测试嵌套括号表达式
    int nested = ((a + b) * 2) - c;  // ((10 + 20) * 2) - 10 = 50
    
    // 测试括号表达式在复杂表达式中
    int sum = (a == c) + (a != b) + (a < b);  // 1 + 1 + 1 = 3
    
    // 测试括号表达式在条件判断中
    if ((a == c)) {
        int condition_result = (a + b) * 2;  // (10 + 20) * 2 = 60
        return sum;  // 应该返回 3
    }
    
    return 0;
}
