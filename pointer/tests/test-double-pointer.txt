// 二级指针操作测试用例
// 测试功能：
// 1. 二级指针声明：int **pp;
// 2. 一级指针声明：int *p;
// 3. 二级指针赋值：pp = &p;
// 4. 通过二级指针访问一级指针：*pp = &a;
// 5. 通过二级指针修改值：**pp = 123;
// 6. 二级指针作为函数参数传递

int modify(int **pp, int value) {
  **pp = value;  // 通过二级指针修改值
  return;
}

int main() {
    int a = 1;
    int *p;
    int **pp;
    
    p = &a;      // p 指向 a
    pp = &p;     // pp 指向 p
    
    **pp = 10;   // 通过二级指针修改 a：a = 10
    
    modify(pp, 100);  // 调用 modify，通过二级指针修改 a：a = 100
    
    return a;  // 应该返回 100
}
