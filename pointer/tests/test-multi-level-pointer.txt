// 多级指针操作测试用例（支持任意级别：1-8级）
// 测试功能：
// 1. 八级指针声明：int ********p8;
// 2. 七级指针声明：int *******p7;
// 3. 六级指针声明：int ******p6;
// 4. 五级指针声明：int *****p5;
// 5. 四级指针声明：int ****p4;
// 6. 三级指针声明：int ***p3;
// 7. 二级指针声明：int **p2;
// 8. 一级指针声明：int *p1;
// 9. 多级指针赋值：p8 = &p7, p7 = &p6, ... p1 = &a;
// 10. 通过1-8级指针修改值：*p1 = 2, **p2 = *p1 + 3, ***p3 = **p2 + 4, ... ********p8 = *******p7 + 9;
// 11. 乘法运算：*（如 2 * 3）
// 12. 幂运算：**（如 2 ** 3）

int main() {
    int a = 1;
    int *p1;
    int **p2;
    int ***p3;
    int ****p4;
    int *****p5;
    int ******p6;
    int *******p7;
    int ********p8;
    
    p1 = &a;    // p1 指向 a
    p2 = &p1;   // p2 指向 p1
    p3 = &p2;   // p3 指向 p2
    p4 = &p3;   // p4 指向 p3
    p5 = &p4;   // p5 指向 p4
    p6 = &p5;   // p6 指向 p5
    p7 = &p6;   // p7 指向 p6
    p8 = &p7;   // p8 指向 p7
    
    // 测试乘法和幂运算
    int x = 2 * 3;      // 乘法：x = 6
    int y = 2 ** 3;     // 幂运算：y = 8
    int z = x * y;      // 乘法：z = 6 * 8 = 48
    
    // 测试不同级别的指针解引用赋值
    *p1 = 2;                    // 通过一级指针修改 a：a = 2
    **p2 = *p1 + 3;            // 通过二级指针修改 a：a = 2 + 3 = 5
    ***p3 = **p2 + 4;           // 通过三级指针修改 a：a = 5 + 4 = 9
    ****p4 = ***p3 + 5;         // 通过四级指针修改 a：a = 9 + 5 = 14
    *****p5 = ****p4 + 6;        // 通过五级指针修改 a：a = 14 + 6 = 20
    ******p6 = *****p5 + 7;      // 通过六级指针修改 a：a = 20 + 7 = 27
    *******p7 = ******p6 + 8;    // 通过七级指针修改 a：a = 27 + 8 = 35
    ********p8 = *******p7 + 9;  // 通过八级指针修改 a：a = 35 + 9 = 44
    
    // 再次测试乘法和幂运算
    int w = a * 2;      // 乘法：w = 44 * 2 = 88
    int v = 3 ** 2;     // 幂运算：v = 9
    
    return a;  // 应该返回 44
}
