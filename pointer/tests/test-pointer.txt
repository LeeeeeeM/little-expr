// 指针操作测试用例
// 测试功能：
// 1. 指针声明：int *p;
// 2. 取地址操作：p = &a;
// 3. 解引用赋值：*p = 123;
// 4. 解引用读取：int x = *p;
// 5. 指针作为函数参数传递

int add(int *p, int a) {
  int b = *p;
  *p = a + b;
  return;
}

int main() {
    int a = 1;
    int *p;
    p = &a;
    *p = 123;  // 通过指针修改 a，此时 a = 123
    add(p, 100);  // 调用 add，通过指针修改 a：*p = a + *p = 100 + 123 = 223
    return a;  // 应该返回 223
}
