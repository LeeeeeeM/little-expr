// 指针操作测试用例
// 测试功能：
// 1. 指针声明：int *p;
// 2. 取地址操作：p = &a;
// 3. 解引用赋值：*p = 123;
// 4. 解引用读取：int x = *p;
// 5. 指针作为函数参数传递

int add(int *p, int a) {
  *p = a + *p;
  return;
}

int main() {
    int a = 1;
    int *p;
    p = &a;
    *p = 123;
    add(p, 100);  // 调用 add，通过指针修改 a：a = a + 100 = 1 + 100 = 101
    return a;  // 应该返回 101
}
