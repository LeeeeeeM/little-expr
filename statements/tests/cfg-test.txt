// CFG测试用例

function testFunction() {
    let x = 10;
    let y = 20;
    
    if (x > 5) {
        y = x * 2;
        return y;
    } else {
        y = x + 1;
        return y;
    }
    
    // 死代码 - 永远不会执行
    let z = 100;
    z = z + 1;
}

function loopFunction() {
    let sum = 0;
    let i = 0;
    
    while (i < 10) {
        sum = sum + i;
        i = i + 1;
    }
    
    return sum;
}

function complexFunction() {
    let a = 1;
    let b = 2;
    
    if (a > 0) {
        if (b > 0) {
            return a + b;
        } else {
            return a - b;
        }
    } else {
        return 0;
    }
    
    // 死代码
    let c = 999;
    c = c * 2;
}
function forLoopFunction() {
    let total = 0;
    let count = 0;
    
    for (let i = 0; i < 5; i = i + 1) {
        total = total + i;
        count = count + 1;
    }
    
    return total;
}
