# 复杂程序测试用例

## 计算器程序
int x = 10;
int y = 5;
print(x + y);
print(x - y);
print(x * y);
print(x / y);
print(x ** y);

## 数学计算
int a = 15;
int b = 3;
int c = a % b;
print(c);

## 条件逻辑
int score = 85;
if (score >= 90) {
    print(1);
} else if (score >= 80) {
    print(2);
} else if (score >= 70) {
    print(3);
} else {
    print(4);
}

## 循环计算
int sum = 0;
int i = 1;
while (i <= 5) {
    sum = sum + i;
    i = i + 1;
}
print(sum);

## 复杂表达式
int result = (1 + 2) * (3 + 4) - 5;
int power = 2 ** 3;
int modulo = 10 % 3;
print(result);
print(power);
print(modulo);
