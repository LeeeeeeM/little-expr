# 综合测试：for、while、块级作用域、let
int main() {
  int sum = 0;
  int i = 0;
  
  # while 循环测试
  while (i < 5) {
    let temp = i * 2;  # let 变量在块级作用域
    sum = sum + temp;
    i = i + 1;
  }
  
  print(sum);  # 应该输出 20 (0+2+4+6+8)
  
  # for 循环测试
  for (int j = 0; j < 3; j = j + 1) {
    let multiplier = j + 1;  # let 变量在 for 循环块中
    int product = sum * multiplier;
    
    # 嵌套块级作用域
    if (product > 30) {
      let bonus = 10;  # let 变量在 if 块中
      product = product + bonus;
    }
    
    print(product);
  }
  
  # 测试 TDZ - 这应该会报错
  # print(undefined_var);  # 取消注释会报错
  
  return 0;
}
