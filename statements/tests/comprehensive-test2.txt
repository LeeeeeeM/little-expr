# 综合测试：for、while、块级作用域、let
int main() {
  int sum = 0;
  int i = 0;
  
  # while 循环测试
  while (i < 5) {
    int temp = i * 2;  # 改为int变量
    sum = sum + temp;
    i = i + 1;
  }
  
  print(sum);  # 应该输出 20 (0+2+4+6+8)
  
  # for 循环测试
  for (int j = 0; j < 3; j = j + 1) {
    int multiplier = j + 1;  # 改为int变量
    int product = sum * multiplier;
    
    # 嵌套块级作用域
    if (product > 30) {
      int bonus = 10;  # 改为int变量
      product = product + bonus;
    }
    
    print(product);
  }
  
  return 0;
}
