# 控制流语句测试用例

## 条件语句
int x = 5;
if (x > 0) {
    print(1);
}

if (x > 0) {
    print(1);
} else {
    print(0);
}

if (x > 0) {
    print(1);
} else if (x < 0) {
    print(-1);
} else {
    print(0);
}

## 循环语句
int x = 3;
while (x > 0) {
    print(x);
    x = x - 1;
}

## 简单循环测试
int count = 0;
while (count < 3) {
    count = count + 1;
    print(count);
}

## 嵌套条件
int score = 85;
if (score > 80) {
    print(1);
    if (score > 90) {
        print(2);
    }
} else {
    print(0);
}
